    Class Log

    Logging utilities

    Setup example: 

    // use stderror for logging
setStderrLogger();
// set log level
setLogLevel(LogLeve.Debug);

    Logging example: 

    // log debug message
Log.d("mouse clicked at ", x, ",", y);
// log error message
Log.e("exception while reading file", e);
